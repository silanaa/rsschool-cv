Odd-Even String Sort KATA from CODEWARS:

const sortMyString = (S) => {
let even = S.split('')
            .filter((v, i) => i % 2 === 0)
            .join('');
let odd = S.split('')
           .filter((v, i) => i % 2 !== 0)
           .join('');
return even + ' ' + odd;
}
